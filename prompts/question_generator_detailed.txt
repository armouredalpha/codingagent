import rclpy
from rclpy.node import Node
from std_srvs.srv import SetBool


class PartsVerifier(Node):
    def __init__(self):
        super().__init__("assembly_station")
        self.srv = self.create_service(SetBool, "/parts_verify", self._verify_cb)
        self.get_logger().info("Parts verification service ready")

    def _verify_cb(self, request, response):
        state = "enabled" if request.data else "disabled"
        self.get_logger().info(f"Verification {state}")
        response.success = True
        response.message = f"Verification {state}"
        return response


def main(args=None):
    rclpy.init(args=args)
    node = PartsVerifier()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()


if __name__ == "__main__":
    main()
