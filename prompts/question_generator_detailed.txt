import rclpy
from rclpy.node import Node
from std_srvs.srv import SetBool


class PartsVerifier(Node):
    """Service server for assembly station parts verification."""

    def __init__(self):
        super().__init__("assembly_station")
        self.srv = self.create_service(
            SetBool, "/parts_verify", self.verify_callback
        )
        self.get_logger().info("Parts verification service started")

    def verify_callback(self, request, response):
        if request.data:
            self.get_logger().info("Verification enabled")
            response.success = True
            response.message = "Verification enabled"
        else:
            self.get_logger().info("Verification disabled")
            response.success = True
            response.message = "Verification disabled"
        return response


def main(args=None):
    rclpy.init(args=args)
    node = PartsVerifier()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()


if __name__ == "__main__":
    main()
